Title Local Reactions over Time
Graph_Subgroup General Principles
Code Added Yes
Description The mean profile of each treatment group is plotted over time for each of the five parameters being collected to assess local reactions to treatment.
Contributor/Email Mat Soukup (email: Mat.Soukup@fda.hhs.gov)
Background Velac Gel is a combination product of which it is well-established that one of it's components, tretinoin, is somewhat of a local irritant. Velac was not approved based on a positive signal in the non-clinical carc study.

library(lattice)

dat <- sasxport.get("//Cdsesub1/n50803/N_000/2004-08-23/crt/datasets/vlc.c.304/A_EFF.XPT")<br /> #Make missing be denoted as NA
levels(dat$dry)[levels(dat$dry)%in%""] <- NA<br /> levels(dat$eryth)[levels(dat$eryth)%in%""] <- NA<br /> levels(dat$scal)[levels(dat$scal)%in%""] <- NA<br /> levels(dat$burn)[levels(dat$burn)%in%""] <- NA<br /> levels(dat$itch)[levels(dat$itch)%in%""] <- NA<br /> dat$dryn <- as.numeric(dat$dry)<br /> dat$erythn <- as.numeric(dat$eryth)<br /> dat$scaln <- as.numeric(dat$scal)<br /> dat$burnn <- as.numeric(dat$burn)<br /> dat$itchn <- as.numeric(dat$itch)<br />
skinval <- with(dat, c(dryn, erythn, scaln, burnn, itchn))<br /> skinpar <- rep(c('Dryness','Erythema','Scaling','Burning','Itching'), each=dim(dat)[1])<br /> skinvisit <- rep(dat$visit, 5)<br /> skintrt <- rep(dat$txname, 5)<br />
ss <- summarize(skinval, llist(skinpar, skintrt, skinvisit), function(x)mean(x, na.rm=TRUE))<br />
#png('C:/Research/Graphics/Graphs4Display/webpages/classes/pages/images/local50803.png',

new.back <- trellis.par.get("background")<br />new.back$col <- "white"<br />newcol <- trellis.par.get("superpose.symbol")<br />newcol$col <- c('blue','green4','black','red' )<br />newcol$pch <- c(16,17,4,8)<br />new.line <- trellis.par.get("superpose.line")<br />new.line$col <- c('blue','green4','black','red' )<br />new.line$lty <- c(1,1,1,1)<br />new.pan <- trellis.par.get("strip.background")<br />new.pan$col <- c('gray90','white')<br />trellis.par.set("background", new.back)
trellis.par.set("superpose.symbol", newcol)
trellis.par.set("superpose.line", new.line)
trellis.par.set("strip.background",new.pan)

xyplot(skinval ~ skinvisit|skinpar, groups=skintrt, data=ss, type='o',
